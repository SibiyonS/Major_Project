import argparse
from pathlib import Path
import pandas as pd
import torch
from sklearn.metrics import accuracy_score, f1_score, precision_score, recall_score, roc_auc_score
from torch.optim import AdamW
from torch.utils.data import DataLoader
from tqdm import tqdm
from torch.amp import autocast, GradScaler
from src.config import AudioConfig, MODELS_DIR, SPLITS_DIR, TrainConfig
from src.data import MelDataset, build_feature_cache, collect_audio_files, make_splits
from src.model import CnnBiLstmDetector
from src.utils import save_json, seed_everything


def run_epoch(model, loader, criterion, optimizer, device, scaler):
    model.train()
    total_loss = 0

    for x, y in tqdm(loader, desc="train", leave=False):
        x, y = x.to(device, non_blocking=True), y.to(device, non_blocking=True)

        optimizer.zero_grad()

        with autocast(device_type=device.type):
            logits = model(x)
            loss = criterion(logits, y)

        scaler.scale(loss).backward()
        scaler.step(optimizer)
        scaler.update()

        total_loss += loss.item() * x.size(0)

    return total_loss / len(loader.dataset)


@torch.no_grad()
def evaluate(model, loader, criterion, device):
    model.eval()
    losses, probs, labels = [], [], []

    for x, y in tqdm(loader, desc="val", leave=False):
        x, y = x.to(device, non_blocking=True), y.to(device, non_blocking=True)

        with autocast(device_type=device.type):
            logits = model(x)
            loss = criterion(logits, y)

        p = torch.sigmoid(logits)

        losses.extend([loss.item()] * x.size(0))
        probs.extend(p.cpu().numpy().tolist())
        labels.extend(y.cpu().numpy().tolist())

    preds = [1 if p >= 0.5 else 0 for p in probs]

    metrics = {
        "loss": float(sum(losses) / len(losses)),
        "accuracy": accuracy_score(labels, preds),
        "precision": precision_score(labels, preds, zero_division=0),
        "recall": recall_score(labels, preds, zero_division=0),
        "f1": f1_score(labels, preds, zero_division=0),
        "roc_auc": roc_auc_score(labels, probs) if len(set(labels)) > 1 else 0.0,
    }

    return metrics


def main(args):
    seed_everything(args.seed)

    # ✅ Proper device
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    print("Using device:", device)

    audio_cfg = AudioConfig()
    train_cfg = TrainConfig(
        batch_size=args.batch_size,
        epochs=args.epochs,
        lr=args.lr,
        hidden_size=args.hidden_size,
        dropout=args.dropout,
        seed=args.seed,
    )

    dataset_root = Path(args.dataset_root)

    FOLDER_LABEL_MAP = {
        "real": 0, "fake": 1,
        "FakeAudios": 1, "RealAudios": 0,
        "human": 0, "nonhuman": 1,
        "possible": 0, "REAL": 0, "FAKE": 1
    }

    index_df = collect_audio_files(dataset_root, FOLDER_LABEL_MAP)

    cached_df = build_feature_cache(index_df, Path(args.cache_dir), audio_cfg)

    train_df, val_df, test_df = make_splits(cached_df)

    SPLITS_DIR.mkdir(parents=True, exist_ok=True)
    train_df.to_csv(SPLITS_DIR / "train.csv", index=False)
    val_df.to_csv(SPLITS_DIR / "val.csv", index=False)
    test_df.to_csv(SPLITS_DIR / "test.csv", index=False)

    # ✅ Better DataLoader
    train_loader = DataLoader(
        MelDataset(train_df, augment=True),
        batch_size=train_cfg.batch_size,
        shuffle=True,
        num_workers=2,
        pin_memory=True
    )

    val_loader = DataLoader(
        MelDataset(val_df, augment=False),
        batch_size=train_cfg.batch_size,
        shuffle=False,
        num_workers=2,
        pin_memory=True
    )

    model = CnnBiLstmDetector(
        hidden_size=train_cfg.hidden_size,
        dropout=train_cfg.dropout
    ).to(device)

    criterion = torch.nn.BCEWithLogitsLoss()
    optimizer = AdamW(model.parameters(), lr=train_cfg.lr, weight_decay=train_cfg.weight_decay)

    # ✅ Mixed precision scaler
    scaler = GradScaler()

    history = []
    best_f1 = -1
    best_path = MODELS_DIR / "best_cnn_bilstm.pt"

    for epoch in range(1, train_cfg.epochs + 1):
        train_loss = run_epoch(model, train_loader, criterion, optimizer, device, scaler)
        val_metrics = evaluate(model, val_loader, criterion, device)

        val_metrics.update({"epoch": epoch, "train_loss": train_loss})
        history.append(val_metrics)

        print(
            f"Epoch {epoch}/{train_cfg.epochs} | "
            f"train_loss={train_loss:.4f} | val_f1={val_metrics['f1']:.4f} | "
            f"val_auc={val_metrics['roc_auc']:.4f}"
        )

        if val_metrics["f1"] > best_f1:
            best_f1 = val_metrics["f1"]
            torch.save(
                {"model_state": model.state_dict(), "config": train_cfg.__dict__},
                best_path
            )

    pd.DataFrame(history).to_csv(MODELS_DIR / "train_history.csv", index=False)
    save_json({"best_f1": best_f1, "device": str(device)}, MODELS_DIR / "train_summary.json")

    print(f"Training done. Best model saved at: {best_path}")

    parser = argparse.ArgumentParser(description="Train CNN-BiLSTM deepfake voice detector")

    parser.add_argument("--dataset_root", type=str, default=r"S:\Major_Project_New\dataset_root")
    parser.add_argument("--cache_dir", type=str, default="artifacts/features")
    parser.add_argument("--epochs", type=int, default=50)

    # ✅ Increased batch size (adjust if memory error)
    parser.add_argument("--batch_size", type=int, default=1)

    parser.add_argument("--lr", type=float, default=1e-3)
    parser.add_argument("--hidden_size", type=int, default=128)
    parser.add_argument("--dropout", type=float, default=0.3)
    parser.add_argument("--seed", type=int, default=42)

    main(parser.parse_args())
