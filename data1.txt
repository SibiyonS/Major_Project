AUDIO_EXTENSIONS = {".wav", ".mp3", ".flac", ".m4a", ".ogg"}


def _infer_label_and_source(path: Path, folder_label_map: dict[str, int] | None = None) -> tuple[int | None, str]:
    """Infer label from any parent folder level.

    Supports deep paths like:
      dataset/fake/speaker1/voice.wav
      dataset/real/speaker2/voice.wav
      dataset/OpenAI/spk1/file.wav

    Priority:
      1) explicit folder_label_map match on any path part
      2) keyword fallback: any part containing "real" -> 0, "fake" -> 1
    """
    folder_label_map = folder_label_map or {}
    map_lower = {k.lower(): v for k, v in folder_label_map.items()}

    parts_lower = [part.lower() for part in path.parts]
    source = path.parent.name

    for part in reversed(parts_lower):
        if part in map_lower:
            return map_lower[part], part

    for part in reversed(parts_lower):
        if "real" in part:
            return 0, part
        if "fake" in part:
            return 1, part

    return None, source


def collect_audio_files(dataset_root: Path, folder_label_map: dict[str, int] | None = None) -> pd.DataFrame:
    rows = []
    for path in dataset_root.rglob("*"):
        if not path.is_file() or path.suffix.lower() not in AUDIO_EXTENSIONS:
            continue

        label, source = _infer_label_and_source(path, folder_label_map)
        if label is None:
            continue

        rows.append({"audio_path": str(path), "label": int(label), "source": source})

    if not rows:
        raise ValueError(f"No labeled audio files found under {dataset_root}")

    df = pd.DataFrame(rows)
    return df.sample(frac=1.0, random_state=42).reset_index(drop=True)

def load_and_preprocess_audio(audio_path: str, cfg: AudioConfig) -> List[np.ndarray]:
    waveform, _ = librosa.load(audio_path, sr=cfg.sample_rate, mono=True)
    waveform, _ = librosa.effects.trim(waveform, top_db=30)

    waveform = waveform / (np.max(np.abs(waveform)) + 1e-8)

    window_size = cfg.max_samples          # e.g. 3 sec
    hop_size = window_size // 2            # 50% overlap

    features = []

    for start in range(0, len(waveform) - window_size + 1, hop_size):
        chunk = waveform[start:start + window_size]

        mel = librosa.feature.melspectrogram(
            y=chunk,
            sr=cfg.sample_rate,
            n_fft=cfg.n_fft,
            hop_length=cfg.hop_length,
            n_mels=cfg.n_mels,
            fmin=cfg.fmin,
            fmax=cfg.fmax,
            power=2.0,
        )

        log_mel = librosa.power_to_db(mel, ref=np.max)
        log_mel = (log_mel - np.mean(log_mel)) / (np.std(log_mel) + 1e-8)

        features.append(log_mel.astype(np.float32))

    # 🔥 handle very short audio
    if len(features) == 0:
        chunk = waveform[:window_size]
        chunk = np.pad(chunk, (0, max(0, window_size - len(chunk))))

        mel = librosa.feature.melspectrogram(
            y=chunk,
            sr=cfg.sample_rate,
            n_fft=cfg.n_fft,
            hop_length=cfg.hop_length,
            n_mels=cfg.n_mels,
            fmin=cfg.fmin,
            fmax=cfg.fmax,
            power=2.0,
        )

        log_mel = librosa.power_to_db(mel, ref=np.max)
        log_mel = (log_mel - np.mean(log_mel)) / (np.std(log_mel) + 1e-8)

        features.append(log_mel.astype(np.float32))

    return features

def build_feature_cache(df: pd.DataFrame, cache_dir: Path, cfg: AudioConfig) -> pd.DataFrame:
    cache_dir.mkdir(parents=True, exist_ok=True)
    out_rows = []

    sample_id = 0

    for _, row in tqdm(df.iterrows(), total=len(df), desc="Preprocessing"):
        features_list = load_and_preprocess_audio(row["audio_path"], cfg)

        for i, feature in enumerate(features_list):
            out_path = cache_dir / f"sample_{sample_id:07d}.npy"
            np.save(out_path, feature)

            out_rows.append(
                {
                    "feature_path": str(out_path),
                    "label": int(row["label"]),
                    "source": row["source"],
                    "audio_path": row["audio_path"],
                }
            )

            sample_id += 1

    return pd.DataFrame(out_rows)

def make_splits(df: pd.DataFrame, val_size: float = 0.15, test_size: float = 0.15) -> Tuple[pd.DataFrame, pd.DataFrame, pd.DataFrame]:
    train_df, temp_df = train_test_split(
        df,
        test_size=(val_size + test_size),
        random_state=42,
        stratify=df["label"],
    )
    relative_test = test_size / (val_size + test_size)
    val_df, test_df = train_test_split(
        temp_df,
        test_size=relative_test,
        random_state=42,
        stratify=temp_df["label"],
    )
    return train_df.reset_index(drop=True), val_df.reset_index(drop=True), test_df.reset_index(drop=True)


class MelDataset(Dataset):
    def __init__(self, df: pd.DataFrame, augment: bool = False):
        self.df = df.reset_index(drop=True)
        self.augment = augment

    def __len__(self) -> int:
        return len(self.df)

    def _augment(self, x: np.ndarray) -> np.ndarray:
        if np.random.rand() < 0.5:
            t = np.random.randint(0, max(1, x.shape[1] // 10))
            x[:, :t] = 0
        if np.random.rand() < 0.5:
            f = np.random.randint(0, max(1, x.shape[0] // 10))
            x[:f, :] = 0
        return x

    def __getitem__(self, idx: int):
        row = self.df.iloc[idx]
        x = np.load(row["feature_path"]).astype(np.float32)
        if self.augment:
            x = self._augment(x.copy())
        x = torch.tensor(x).unsqueeze(0)
        y = torch.tensor(float(row["label"]), dtype=torch.float32)
        return x, y
